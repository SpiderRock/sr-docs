SyntheticExpiryQuote

V8 Message Definiton

Live synthetic expiry quotes are SpiderRock option/expiry underlier quotes. These can be from synthetic/implied futures quotes or other sources

CREATE TABLE EXAMPLE QUERY​

CREATE TABLE `SRLive`.`MsgSyntheticExpiryQuote` (
    `ekey_at` ENUM('None','EQT','IDX','BND','CUR','COM','FUT','SYN','WAR','FLX','MUT','SPD','MM','MF','COIN','TOKEN','ANY') NOT NULL DEFAULT 'None' COMMENT 'root + expiry',
    `ekey_ts` ENUM('None','SR','NMS','CME','ICE','CFE','CBOT','NYMEX','COMEX','RUT','CIDX','ARCA','NYSE','OTC','NSDQ','MFQS','MIAX','DJI','CUSIP','ISIN','BXE','ESX','ANY','CXE','DXE','NXAM','NXBR','NXDUB','NXLS','NXLDN','NXML','NXMLT','NXOS','NXP','EUREX','CEDX','ICEFE') NOT NULL DEFAULT 'None' COMMENT 'root + expiry',
    `ekey_tk` VARCHAR(12) NOT NULL DEFAULT '' COMMENT 'root + expiry',
    `ekey_yr` SMALLINT UNSIGNED NOT NULL DEFAULT 0 COMMENT 'root + expiry',
    `ekey_mn` TINYINT UNSIGNED NOT NULL DEFAULT 0 COMMENT 'root + expiry',
    `ekey_dy` TINYINT UNSIGNED NOT NULL DEFAULT 0 COMMENT 'root + expiry',
    `fairPrice` DOUBLE NOT NULL DEFAULT 0,
    `bidPrice` DOUBLE NOT NULL DEFAULT 0 COMMENT 'best implied bid price',
    `askPrice` DOUBLE NOT NULL DEFAULT 0 COMMENT 'best implied ask price',
    `bidSize` INT NOT NULL DEFAULT 0 COMMENT 'best implied bid size',
    `askSize` INT NOT NULL DEFAULT 0 COMMENT 'best implied ask size',
    `bidSource` ENUM('None','Future','FutureRoll','Stock') NOT NULL DEFAULT 'None',
    `askSource` ENUM('None','Future','FutureRoll','Stock') NOT NULL DEFAULT 'None',
    `marketStatus` ENUM('None','PreOpen','PreCross','Cross','Open','Closed','Halted','AfterHours') NOT NULL DEFAULT 'None' COMMENT 'composite market status of all dependent markets',
    `netTimestamp` BIGINT NOT NULL DEFAULT 0 COMMENT 'most recent net timestamp of any market affecting this record',
    PRIMARY KEY USING HASH (`ekey_tk`,`ekey_yr`,`ekey_mn`,`ekey_dy`,`ekey_at`,`ekey_ts`)
) ENGINE=SRSE DEFAULT CHARSET=LATIN1 COMMENT='Live synthetic expiry quotes are SpiderRock option/expiry underlier quotes.  These can be from synthetic/implied futures quotes or other sources';

SELECT TABLE EXAMPLE QUERY​

SELECT
    `ekey_at`,
    `ekey_ts`,
    `ekey_tk`,
    `ekey_yr`,
    `ekey_mn`,
    `ekey_dy`,
    `fairPrice`,
    `bidPrice`,
    `askPrice`,
    `bidSize`,
    `askSize`,
    `bidSource`,
    `askSource`,
    `marketStatus`,
    `netTimestamp`
FROM `SRLive`.`MsgSyntheticExpiryQuote`
WHERE 
    /* Replace with a ENUM('None','EQT','IDX','BND','CUR','COM','FUT','SYN','WAR','FLX','MUT','SPD','MM','MF','COIN','TOKEN','ANY') */ 
    `ekey_at` = 'None'
  AND
    /* Replace with a ENUM('None','SR','NMS','CME','ICE','CFE','CBOT','NYMEX','COMEX','RUT','CIDX','ARCA','NYSE','OTC','NSDQ','MFQS','MIAX','DJI','CUSIP','ISIN','BXE','ESX','ANY','CXE','DXE','NXAM','NXBR','NXDUB','NXLS','NXLDN','NXML','NXMLT','NXOS','NXP','EUREX','CEDX','ICEFE') */ 
    `ekey_ts` = 'None'
  AND
    /* Replace with a VARCHAR(12) */ 
    `ekey_tk` = 'Example_ekey_tk'
  AND
    /* Replace with a SMALLINT UNSIGNED */ 
    `ekey_yr` = 123
  AND
    /* Replace with a TINYINT UNSIGNED */ 
    `ekey_mn` = 1
  AND
    /* Replace with a TINYINT UNSIGNED */ 
    `ekey_dy` = 1;
