CREATE TABLE EXAMPLE QUERY​

CREATE TABLE `SRRisk`.`MsgCurrencyTransaction` (
    `transactionID` BIGINT NOT NULL DEFAULT 0,
    `tradeDate` DATE NOT NULL DEFAULT '1900-01-01',
    `accnt` VARCHAR(16) NOT NULL DEFAULT '',
    `clientFirm` VARCHAR(16) NOT NULL DEFAULT '',
    `botCurrency` ENUM('None','AUD','BRL','CAD','CHF','CNH','CNY','EUR','GBP','JPY','KRW','MXN','MYR','NOK','NZD','SEK','TRY','USD','USDCents','CZK','ZAR','HUF','USX','GBX','DKK','GEL') NOT NULL DEFAULT 'None',
    `botQuantity` DOUBLE NOT NULL DEFAULT 0,
    `sldCurrency` ENUM('None','AUD','BRL','CAD','CHF','CNH','CNY','EUR','GBP','JPY','KRW','MXN','MYR','NOK','NZD','SEK','TRY','USD','USDCents','CZK','ZAR','HUF','USX','GBX','DKK','GEL') NOT NULL DEFAULT 'None',
    `sldQuantity` DOUBLE NOT NULL DEFAULT 0,
    `comment` TINYTEXT NOT NULL DEFAULT '',
    `transactionDttm` DATETIME(6) NOT NULL DEFAULT '1900-01-01 00:00:00.000000',
    `timestamp` DATETIME(6) NOT NULL DEFAULT '1900-01-01 00:00:00.000000',
    PRIMARY KEY USING HASH (`transactionID`,`tradeDate`)
) ENGINE=SRSE DEFAULT CHARSET=LATIN1 COMMENT='';
